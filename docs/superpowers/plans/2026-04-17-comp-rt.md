# COMP-RT Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Fix three Compose runtime gaps: event coalescing (RT-1), client hydration on connect (RT-2), and connector trait split (RT-3).

**Architecture:** A new `CoalescingBuffer` utility replaces the ad-hoc 100ms debounce in `VisionServer` and adds buffered dispatch to agent SSE. On connect, WS clients receive a `hydrate` snapshot; SSE clients receive a replay of the last 50 messages. Connector interfaces are split into `ConnectorDiscovery` + `ConnectorRuntime` reference contracts, with stubs added to `AgentConnector`.

**Tech Stack:** Node.js 22 built-in test runner (`node --test`), ES modules, WebSocket (`ws`), EventSource (browser), Zustand (frontend).

**Design doc:** `compose/docs/features/COMP-RT/design.md`

```js
// compose/server/coalescing-buffer.js

export class CoalescingBuffer {
  #modes = new Map();   // key → 'latest-wins' | 'append'
  #pending = new Map(); // key → value | value[]
  #timer = null;
  #flushFn;

  constructor(flushFn, { intervalMs = 16 } = {}) {
    this.#flushFn = flushFn;
    this.#timer = setInterval(() => this.#flush(), intervalMs);
  }

  register(key, mode) {
    if (mode !== 'latest-wins' && mode !== 'append') {
      throw new Error(`Unknown mode: ${mode}`);
    }
    this.#modes.set(key, mode);
  }

  put(key, value) {
    const mode = this.#modes.get(key);
    if (!mode) throw new Error(`Key not registered: ${key}`);
    if (mode === 'latest-wins') {
      this.#pending.set(key, value);
    } else {
      const arr = this.#pending.get(key) ?? [];
      arr.push(value);
      this.#pending.set(key, arr);
    }
  }

  stop() {
    if (this.#timer) {
      clearInterval(this.#timer);
      this.#timer = null;
    }
  }

  #flush() {
    if (this.#pending.size === 0) return;
    const data = Object.fromEntries(this.#pending);
    this.#pending.clear();
    this.#flushFn(data);
  }
}
```

```js
// Helper: collect N lines from an SSE response
function collectSSELines(port, path, lineCount, timeoutMs = 2000) {
  return new Promise((resolve, reject) => {
    const lines = [];
    const req = http.get({ hostname: 'localhost', port, path, headers: { Accept: 'text/event-stream' } }, (res) => {
      let buf = '';
      res.on('data', (chunk) => {
        buf += chunk.toString();
        const parts = buf.split('\n\n');
        buf = parts.pop();
        for (const part of parts) {
          if (part.trim()) lines.push(part);
          if (lines.length >= lineCount) { req.destroy(); resolve(lines); }
        }
      });
    });
    req.on('error', (e) => { if (e.code !== 'ECONNRESET') reject(e); });
    setTimeout(() => { req.destroy(); resolve(lines); }, timeoutMs);
  });
}
```
